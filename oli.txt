
  const bookingSchema = new Schema({
    name: String,
    orderDate: Date,
    checkIn: Date,
    checkOut: Date,
    status: Boolean
  })

  const userSchema = new Schema ({
    name: String,
    email: String,
    startDate: Date,
    description: String,
    phone: Number,
    status: Boolean,
    password: String
  })

  const contactSchema = new Schema({
    name: String,
    email: String,
    phone: Number,
    date: Date,
    subject: String
  })


  export const User = model('User', userSchema)
